while循环的使用 注意缩进，变量后紧跟着等号，等号后面紧跟着值。


#!/bin/bash
n=0
while [ $n -lt 5 ];do
      let  n=$n+1
   if [ $n -eq 3 ];then
       continue
    fi
    echo $n
done

condition=1
device_number="sasas ssss hjk"
selected_device="hjk"

while [ $condition -eq 1 ];do
   result=$(echo $device_number | grep "${selected_device}")
   echo  $result
   if [[ "$result" != "" ]]
   then
     echo "包含"
     condition=0
   else
    echo "不包含"
    condition=1
   fi
done
